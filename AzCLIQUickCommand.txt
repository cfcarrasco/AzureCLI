#Create a new RG
RGROUP=$(az group create --name vmbackups --location westus2 --output tsv --query name)

#Create a VNET
az network vnet create 
	--resource-group $RGROUP 
	--name NorthwindInternal 
	--address-prefixes 10.0.0.0/16
	--subnet-name NorthwindInternal1 
	--subnet-prefixes 10.0.0.0/24

$Create a VM
az vm create --resource-group vmbackups --name NWAPP01 --size Standard_DS1_v2 --public-ip-sku Standard --vnet-name NorthwindInternal --subnet NorthwindInternal1 --image Win2016DataCenter --admin-username admin123 --no-wait --admin-password "B2tmand@3455"
